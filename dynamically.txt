package com.example.eventlistner2

import android.os.Bundle
import android.view.KeyEvent
import android.view.MotionEvent
import android.widget.Button
import android.widget.EditText
import android.widget.ImageView
import android.widget.TextView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val btnClick = findViewById<Button>(R.id.btnClick)
        val btnLongClick = findViewById<Button>(R.id.btnLongClick)
        val imgTouch = findViewById<ImageView>(R.id.imgTouch)
        val etInput = findViewById<EditText>(R.id.etInput)
        val tvMessage = findViewById<TextView>(R.id.tvMessage)

        // ✅ Button Click Listener
        btnClick.setOnClickListener {
            tvMessage.text = "Button Clicked!"
            Toast.makeText(this, "Button Clicked!", Toast.LENGTH_SHORT).show()
        }

        // ✅ Button Long Click Listener
        btnLongClick.setOnLongClickListener {
            tvMessage.text = "Long Click Detected!"
            Toast.makeText(this, "Long Click Detected!", Toast.LENGTH_SHORT).show()
            true // Returning true means the event is consumed
        }

        // ✅ ImageView Touch Listener (Handles ACTION_UP, ACTION_DOWN, ACTION_POINTER_DOWN)
        imgTouch.setOnTouchListener { _, event ->
            when (event.action) {
                MotionEvent.ACTION_DOWN -> {
                    tvMessage.text = "Action Down Detected!"
                    Toast.makeText(this, "Action Down", Toast.LENGTH_SHORT).show()
                }
                MotionEvent.ACTION_UP -> {
                    tvMessage.text = "Action Up Detected!"
                    Toast.makeText(this, "Action Up", Toast.LENGTH_SHORT).show()
                }
                MotionEvent.ACTION_POINTER_DOWN -> {
                    tvMessage.text = "Action Pointer Down Detected!"
                    Toast.makeText(this, "Action Pointer Down", Toast.LENGTH_SHORT).show()
                }
            }
            true
        }

        // ✅ EditText Key Listener (Handles Enter Key)
        etInput.setOnKeyListener { _, keyCode, event ->
            if (event.action == KeyEvent.ACTION_DOWN && keyCode == KeyEvent.KEYCODE_ENTER) {
                tvMessage.text = "Enter Key Pressed: ${etInput.text}"
                Toast.makeText(this, "You pressed Enter!", Toast.LENGTH_SHORT).show()
                true
            } else {
                false
            }
        }
    }
}
